let curry = (fn) => {
  let limit = fn.length;
  let judgeCurry = (...args) =>
    args.length >= limit
      ? fn.apply(null, args)
      : (...args2) => judgeCurry.apply(null, args.concat(args2));
  return judgeCurry;
};

let afterCurry = curry(add);
let test = afterCurry(1)(2)(3);
