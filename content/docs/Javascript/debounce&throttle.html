let debounce = (func, delay, immediate) => {
  immediate = immediate === void 0 ? true : immediate;
  let lastTime = 0;
  let timer = null;
  return (...arg) => {
    let now = Date.now();
    if (now - lastTime >= delay && immediate) {
      func.apply(this, arg);
    }
    if (!immediate) {
      if (timer) {
        clearTimeout(timer);
      }
      timer = setTimeout(() => {
        func.apply(this, arg);
      }, delay);
    }
    lastTime = now;
  };
};
