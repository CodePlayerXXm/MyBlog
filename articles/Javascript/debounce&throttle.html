            # 代码实现

            let debounce = (func, delay, immediate) => {
  immediate = immediate === void 0 ? true : immediate;
  let lastTime = 0;
  let timer = null;
  return (...arg) => {
    let now = Date.now();
    if (now - lastTime >= delay && immediate) {
      func.apply(this, arg);
    }
    if (!immediate) {
      if (timer) {
        clearTimeout(timer);
      }
      timer = setTimeout(() => {
        func.apply(this, arg);
      }, delay);
    }
    lastTime = now;
  };
};

            func 为要执行的函数，delay 为一定时间，immediate 为布尔值，true 表示立即执行，false 表示过一定间隔后在执行。

            let throttle = (func, delay, immediate) => {
  immediate = immediate === void 0 ? true : immediate;
  let lastTime = 0;
  let timer = null;
  let remain = delay;
  let flag = true;
  return (...arg) => {
    if (immediate) {
      let now = Date.now();
      lastTime = lastTime === void 0 ? now : lastTime;
      remain -= now - lastTime;
      if (remain <= 0) {
        func.apply(this, arg);
        remain = delay;
      }
      lastTime = now;
      return;
    }
    if (!flag) return;
    flag = false;
    setTimeout(() => {
      func.apply(this, arg);
      flag = true;
    }, delay);
  };
};
